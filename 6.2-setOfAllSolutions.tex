\subsection{The set of all solutions to a system of linear equations}
\tsLem{6.2.1 (Injectivity of \(A\) on \(C(A^\top)\), uniqueness of sol.)}{\(A \in \mathbb{R}^{m \times n}, \ x,y\in C(A^\top): \ Ax = Ay \Leftrightarrow x = y \) \\ This leads to: \(C(A^T) \cap N(A) = \{0\}\)}
\newline
\tsThe{6.2.2 (Set of all solution of linear equations)}{Set of all sol. : $\{x \in \mathbb{R}^n | Ax = b\}  \neq \emptyset$, then: \\ $\{ x \in \mathbb{R}^n \mid Ax = b \} = x_1 + N(A), \ \ x_1 \in R(A) \text{ is unique s.t. } Ax_1 = b.$}
\newline
\tsThe{6.2.4 (Linear equations with no solution)}{Linear equations has no solution: \\ $\{ x \in \mathbb{R}^n \mid Ax = b \} = \emptyset \;\Longleftrightarrow\; \{ z \in \mathbb{R}^m \mid A^{T} z = 0,\; b^{T} z = 1 \} \neq \emptyset.$}
